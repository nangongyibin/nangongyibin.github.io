就去C:\Users\XXXX\AppData\Roaming\Tencent\QQMusic下找到CachePath.ini文件，右键编辑-修改Path=F:\Download\Cache\QQCache\QQMusicCache。

	[Cache]
	
	Path=F:\Download\Cache\QQCache\QQMusicCache\
	
	Volumn=1400
	
	Convert=1
