## Log类使用如下: ##

                System.out.println("111111");
                Log.v(TAG, "onClick: 222222");
                Log.d(TAG, "onClick: 333333");
                Log.i(TAG, "onClick: 444444");
                Log.w(TAG, "onClick: 555555");
                Log.e(TAG, "onClick: 666666");
